	    steps {

	        input ('Do you want to proceed')

	 }

	         not {

	              branch "master"

	         steps {

	            echo "Hello"

	       }
	 }

	     Parallel {

	         stage('unit test') {

	           steps {

	           }

	         stage('integrity test'){

	              agent {

	                   docker {

	                         reuseNode True

	                   }
	                steps {

	                }
	              }

	         }
	     }




	 }
